```.py
def mystery(A:int, B:int):
    out = (A * B) - abs(A - B)
    return out

out1 = mystery(A = 2, B = 6)
out2 = mystery(A = 4, B = 10)
out3 = mystery(A = 10, B = 10)
out4 = mystery(A = 70, B = 50)
print(out1, out2, out3,out4)
```
